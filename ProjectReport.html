  if (error.name === "CastError") {
    return response.status(400).send({
      error: "malformatted id",
    });
  } else if (error.name === "ValidationError") {
    return response.status(400).json({
      error: error.message,
    });
  } else if (error.name === "JsonWebTokenError") {
    return response.status(401).json({
      error: "invalid token",
    });
  }

  logger.error(error.message);

  next(error);
};

//token extractor for jwt authentication tokens
const tokenExtractor = async (request, response, next) => {
  const authorization = request.get("authorization");
  if (authorization && authorization.toLowerCase().startsWith("bearer ")) {
    request.token = authorization.substring(7);
  } else {
    request.token = null;
  }
  next();
};

Admin (Abdullah Arif)

  1. User Management:

    • Sign up as an admin: This endpoint allows creating a new user with the role set as “admin”. The user information such as username, fullname, email, password, phone, and address are provided in the request body. The password is hashed using bcrypt before saving it to the database.
    • Edit user: This endpoint allows updating user information. The user is identified by the :id parameter in the URL. The updated information such as username, fullname, email, password, phone, and address is provided in the request body. The password is hashed using bcrypt before updating it in the database.
    • Delete a user: This endpoint allows deleting a user. The user is identified by the :id parameter in the URL. The user can only be deleted if the authenticated user is an admin.

  2. Property Management:

    • Delete a property: This endpoint allows deleting a property. The property is identified by the :id parameter in the URL. The property can only be deleted if the authenticated user is a seller or an admin and if the property is owned by the user.
    • Edit a property: This endpoint allows updating property information. The property is identified by the :id parameter in the URL. The updated information such as title, description, price, photos, location, history, and tax is provided in the request body. The property can only be updated if the authenticated user is a seller or an admin and if the property is owned by the user.
    • Approve a property: This endpoint allows an admin to approve a property. The property is identified by the :id parameter in the URL. The property’s isApproved flag is set to true, indicating that it has been approved by the admin.
    • Verify a property: This endpoint allows an admin to verify the documents of a property. The property is identified by the :id parameter in the URL. The property’s isVerified flag is set to true, indicating that the documents have been verified by the admin.

  3. Messages Management:

    • Find all messages sent by a sender: This endpoint retrieves all messages sent by a particular sender. The sender’s ID is provided as the :sender parameter in the URL. The messages are returned in the response.
    • Find all messages received by a receiver: This endpoint retrieves all messages received by a particular receiver. The receiver’s ID is provided as the :receiver parameter in the URL. The messages are returned in the response.
    • Fetch communication between a sender and receiver: This endpoint retrieves all messages exchanged between a sender and a receiver. The sender’s ID is provided as the :sender parameter, and the receiver’s ID is provided as the :receiver parameter in the URL. The messages are returned in the response.

  4. Appointment Management:

    • Confirm an appointment: This endpoint allows an admin to confirm an appointment. The appointment is identified by the :id parameter in the URL. The appointment’s status is updated to “confirmed” in the database.

These functionalities provide the admin with the ability to manage users, properties, messages, and appointments within the system.

Seller (Tayyba Saveera)

  1. Login/Signup:

    • The /login endpoint allows sellers to log in. It receives the username and password in the request body, queries the database for a matching user record, and compares the provided password with the hashed password stored in the database using bcrypt.
    • If the login is successful, a JSON web token (JWT) is created using the user’s username and _id, signed with the process.env.SECRET value.
    • The response includes the generated token, username, and fullname of the user.

  2. Property Listing, Selling:

    • The /property endpoint allows sellers to list a new property for sale.
    • The endpoint expects fields such as title, description, price, location, history, and tax in the request body.
    • The seller’s authentication is verified by checking the JWT in the request header (request.token) and ensuring the user’s role is set to “seller”.
    • The property details are saved in the database, including the attachments (photos) uploaded by the seller. The file paths are stored in the photos array field.
    • The property is associated with the seller by setting the owner field to the seller’s _id.
    • The saved property is returned in the response.

  3. Messaging:

    • The messaging functionality includes sending simple messages and messages with attachments between users.
    • The /simple endpoint allows users to send a simple text-based message. The sender, receiver, and content fields are expected in the request body.
    • The /attachment endpoint allows users to send messages with attachments. It expects the same fields as the /simple endpoint, along with the attachments uploaded using the attachments field.
    • The file paths of the uploaded attachments are stored in the attachments array field.
    • The messages are saved in the database with the sender’s and receiver’s _id, message content, and the current date and time.
    • Messages can be searched using the /search endpoint, which takes a keyword query parameter and returns messages where the content matches the keyword (case-insensitive).
    • Users can also archive, unarchive, mark as read, and delete messages using the respective endpoints.

  4. Appointment:

    • The appointment functionality allows users to schedule and manage appointments for property viewings.
    • The /appointment endpoint is used to schedule a new appointment. The sellerId, buyerId, and propertyId are expected in the request body.
    • The endpoint verifies the existence of the seller, buyer, and property in the database.
    • It also ensures that the seller is the owner of the property by comparing their _id values.
    • The appointment details, including the current date and time and a default status of “pending,” are saved in the database.
    • Appointments can be canceled by using the /cancel endpoint, which updates the status to “cancelled” for the specified appointment.
    • The /seller/:sellerId endpoint allows sellers to retrieve all their appointments by providing their _id as a URL parameter.
    • The appointments are populated with the buyer’s name and property title for better readability.

Customer (Usman)

  1. Login/Signup:
    The login and signup functionality allows users to create an account or authenticate themselves using their username and password. Here’s an overview of the code implementation:
  • Login: The server receives a POST request with the user’s username and password. It retrieves the user from the database using the provided username and compares the hashed password with the input password using bcrypt. If the authentication is successful, a JSON web token (JWT) is generated and sent back to the client.
  • Signup: The server receives a POST request with the user’s username, fullname, email, password, phone, and address. It checks if the username and email are unique and if the password meets the required criteria. If all conditions are met, the password is hashed using bcrypt and a new user object is created and saved in the database.
  1. Property Search, Wishlist, Buy:
    These functionalities allow customers to search for properties, add them to their wishlist, and buy properties. Here’s an overview of the code implementation:
  • Fetch all properties: The server responds to a GET request to retrieve all properties from the database. The response includes information about the owner of each property.
  • Fetch single property: The server responds to a GET request with a specific property ID to retrieve the details of that property from the database. The response includes information about the owner of the property.
  • Fetch properties by owner: The server responds to a GET request with the owner’s ID to retrieve all properties owned by that specific user.
  • Search properties by keywords in title: The server responds to a GET request with a search query to find properties with titles containing the specified keywords. The response includes information about the owner of each property.
  • Search properties by keywords in title and location: The server responds to a GET request with search keywords and a location to find properties with titles and locations matching the specified criteria. The response includes information about the owner of each property.
  • Buy property: The server receives a PUT request with the property ID to mark the property as sold. It verifies the user’s token, checks if the user exists and has the role of a customer, and updates the property’s buyer and sold status accordingly.
  • Cancel buy process: The server receives a PUT request with the property ID to cancel the purchase process. It verifies the user’s token, checks if the user exists and has the role of a buyer, and updates the property’s buyer and sold status accordingly.
  • Fetch all approved properties: The server responds to a GET request to retrieve all properties that have been approved.
  • Fetch all verified properties: The server responds to a GET request to retrieve all properties that have been verified.
  1. Messaging:

Send a simple message:

  • Route: POST /simple
  • This route allows customers to send a simple text message to another user.
  • It expects the sender, receiver, and content fields in the request body.
  • The endpoint creates a new Message document with the sender, receiver, content, and current date-time.
  • The saved message is returned as the response.

Send a message with attachments:

  • Route: POST /attachment
  • This route enables customers to send a message with attachments (files).
  • It expects the sender, receiver, and content fields in the request body.
  • The attachments are uploaded using the upload.array middleware.
  • The endpoint creates a new Message document with the sender, receiver, content, attachments, and current date-time.
  • The saved message is returned as the response.

Search messages:

  • Route: GET /search
  • Customers can search for messages containing a specific keyword.
  • The keyword is provided as a query parameter.
  • The endpoint uses a regular expression to find messages with content matching the keyword.
  • The matching messages are returned as the response.

Archive a message:

  • Route: PUT /:messageId/archive
  • Customers can archive a specific message by providing its messageId as a parameter.
  • The endpoint updates the isArchived field of the message to true.
  • The updated message is returned as the response.

Unarchive a message:

  • Route: PUT /:messageId/unarchive
  • Customers can unarchive a specific message by providing its messageId as a parameter.
  • The endpoint updates the isArchived field of the message to false.
  • The updated message is returned as the response.

Mark a message as read:

  • Route: PUT /:messageId/read
  • Customers can mark a specific message as read by providing its messageId as a parameter.
  • The endpoint updates the isRead field of the message to true.
  • The updated message is returned as the response.

Delete a message:

  • Route: DELETE /:messageId
  • Customers can delete a specific message by providing its messageId as a parameter.
  • The endpoint finds and deletes the message from the database.
  • A 204 No Content response is sent to indicate a successful deletion.
  1. Appointment:

The appointment routes handle functionalities related to scheduling appointments, canceling appointments, and retrieving appointments.

Schedule an appointment:

  • Route: POST /
  • Customers can schedule an appointment by providing the sellerId, buyerId, and propertyId in the request body.
  • The endpoint verifies the existence of the seller, buyer, and property before creating the appointment.
  • It also ensures that the seller is the owner of the property.
  • The appointment is saved with the current date-time and a status of “pending”.
  • The saved appointment is returned as the response.

Cancel an appointment:

  • Route: PUT /:id/cancel
  • Customers can cancel a specific appointment by providing its id as a parameter.
  • The end point updates the status of the appointment to “cancelled”.
  • The updated appointment is returned as the response.

Get buyer’s appointments:

  • Route: GET /buyer/:buyerId
  • Customers can retrieve appointments associated with a specific buyer by providing the buyerId as a parameter.
  • The endpoint finds appointments where the buyer matches the provided buyerId.
  • The seller and property fields of each appointment are populated with the corresponding user and property data.
  • The appointments are returned as the response.

These routes provide the necessary functionality for customers to communicate through messages and schedule appointments with sellers.

loginRouter.post("/", async (request, response) => {
  const { username, password } = request.body;
  const user = await User.findOne({ username });
  console.log(user);
  const passwordCorrect =
    user === null ? false : await bcrypt.compare(password, user.password);
  console.log(passwordCorrect);
  if (!(user && passwordCorrect)) {
    return response.status(401).json({
      error: "invalid username or password",
    });
  }

  const userForToken = {
    username: user.username,
    id: user._id,
  };

  const token = jwt.sign(userForToken, process.env.SECRET);

  response
    .status(200)
    .send({ token, username: user.username, fullname: user.fullname });
});


//signup 
usersRouter.post("/customer", async (request, response) => {
  try {
    const { username, fullname, email, password, phone, address } =
      request.body;
    const existingUsername = await User.findOne({ username });
    const existingEmail = await User.findOne({ email });
    if (request.body.password.length < 5) {
      return response
        .status(400)
        .json({ error: "Password must be at least 5 characters long" });
    } else if (existingUsername || existingEmail) {
      return response.status(400).json({
        error: "username and email must be unique",
      });
    } else {
      const saltRounds = 10;
      const passwordHash = await bcrypt.hash(password, saltRounds);

      const newUser = new User({
        username,
        fullname,
        email,
        password: passwordHash,
        phone,
        address,
        role: "customer",
      });
      const savedUser = await newUser.save();
      response.status(201).json(savedUser);
    }
  } catch (error) {
    response.status(400).json({ message: error.message });
  }
});
